171  template <typename Container>
172  Array& removeElementsFromContainer(const Container& newElements, bool onlyFirstOcurrence = false) {
173 
174  SizeType elementsFound = 0;
175  SizeType countOfContainer = std::distance(std::begin(newElements), std::end(newElements));
176  std::unique_ptr<SizeType[]> elementsPosition(new SizeType[countOfContainer]);
177 
178  auto newElement = std::begin(newElements);
179 
180  do {
181  elementsFound = 0;
182  newElement = std::begin(newElements);
183 
184  for (SizeType index = 0; index < count_; ++index) {
185 
186  if (values[index] == *newElement) {
187  elementsPosition[elementsFound] = index;
188  elementsFound += 1;
189  ++newElement;
190  }
191  else if (elementsFound != newElements.size()) {
192  elementsFound = 0;
193  newElement = std::begin(newElements);
194  } else { break; }
195  }
196 
197  if (elementsFound == newElements.size()) {
198  for (SizeType i = 0; i < newElements.size(); ++i) {
199  this->removeAt(elementsPosition[0]);
200  }
201  }
202  } while((elementsFound == newElements.size()) && !onlyFirstOcurrence);
203 
204  return *this;
205  }

293  void insert(const Type& newElement, const SizeType index, const SizeType capacityResizeFactor = 2) {
294 
295  if (index != 0) {
296  checkIfOutOfRange(index);
297  }
298  else if (index == count_ || this->isEmpty()) {
299  this->append(newElement);
300  return;
301  }
302 
303  if (capacity_ == count_) {
304 
305  capacity_ *= capacityResizeFactor;
306 
307  Pointer newValues { new Type[capacity_] };
308 
309  std::copy(&values[0], &values[index], &newValues[0]);
310  std::copy(&values[index], &values[count_], &newValues[index+1]);
311 
312  values = std::move(newValues);
313  }
314  else {
315  std::copy(&values[index], &values[count_], &values[index + 1]);
316  }
317 
318  values[index] = newElement;
319  count_ += 1;
320  }
